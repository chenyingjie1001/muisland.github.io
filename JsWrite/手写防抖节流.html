    # 手写防抖节流

    防抖：当触发一个事件，在规定时间内没有触发才执行，如果在规定时间内触发了，则重新计时
    应用：按钮提交只提交最后一次，输入框输完搜索

    function debounce(fn, delay){
   let timer;
   return function(){
       const context = this
       const args = arguments
       if(!timer) {
          setTimeout(function(){
           fn.apply(context, args)
           timer = null
          },delay)
       }
       
   }
}

    节流：在一定时间内，只触发一次
    应用：拖拽触发、浏览resize

    function throttle(fn, delay){
   let prev = 0;
   return function(){
      let now = +new Date()
      let args = arguments
      if(now - prev > delay){
        fn.apply(this, args)
        prev = now
      }
   }
}

    function throttle2(fn, delay){
    let timer
   return function(){
      if(!timer) {
         let args = arguments
         let context = this
         setTimeout(function(){
            fn.apply(this, args)
            timer = null
         }, delay)
      }
   }
}
